\documentclass{article}
\usepackage{tikz}
\usepackage{animate}
\usepackage{float} % Recommended for robust figure placement

\begin{document}

\begin{figure}[H]
    \centering
    \begin{animateinline}[controls, loop, poster=first]{1}

        % Everything below this line is part of a single animation frame/page

        \begin{tikzpicture} % <-- \currentanimframe is defined *inside* here
            \useasboundingbox (0,0) rectangle (25, 4);

            % This conditional checks the frame number set by 'animate'
            \ifnum\currentanimframe=0
                \begin{scope}[shift={(0,0)}]
                    \draw[blue, thick] (0,0) rectangle (4,2);
                    \node at (2,1) {Sketch 1 (Frame 0)};
                \end{scope}
            \fi

            \ifnum\currentanimframe=1
                \begin{scope}[shift={(10,0)}]
                    \draw[red, thick] (0,0) rectangle (4,2);
                    \node at (2,1) {Sketch 2 (Frame 1)};
                \end{scope}
            \fi

            \ifnum\currentanimframe=2
                \begin{scope}[shift={(20,0)}]
                    \draw[green, thick] (0,0) rectangle (4,2);
                    \node at (2,1) {Sketch 3 (Frame 2)};
                \end{scope}
            \fi
        \end{tikzpicture}

        \newframe
        \newframe
        % No newframe after the very last frame

    \end{animateinline}
    \caption{Interactive sequence.}
\end{figure}

\end{document}
